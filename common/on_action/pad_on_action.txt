on_death = { 
	on_actions = { 
		pad_update_candidates_action 
	} 
} 

pad_update_candidates_action = { 
	effect = {
		clear_global_variable_list = pad_candidates

		every_living_character = {
			limit = {
				is_ruler = yes
				has_trait = paranoid
				has_trait = content
			}

			add_to_global_variable_list = {
				name = pad_candidates
				target = this
			}
		}
	} 
}
